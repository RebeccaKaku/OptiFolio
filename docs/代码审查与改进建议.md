# 智能缓存策略
class SmartCache:
    """智能缓存管理器"""
    def __init__(self):
        self._cache = {}
        self._access_stats = {}  # 访问统计
        
    def get(self, key):
        """获取缓存，根据访问频率动态调整TTL"""
        if key in self._cache:
            data, expiry = self._cache[key]
            if time.time() < expiry:
                # 更新访问统计
                self._access_stats[key] = self._access_stats.get(key, 0) + 1
                return data
        return None
        
    def set(self, key, value, base_ttl=3600):
        """设置缓存，热门数据TTL更长"""
        access_count = self._access_stats.get(key, 0)
        # 访问次数越多，TTL越长（最长24小时）
        ttl = min(base_ttl * (1 + math.log2(access_count + 1)), 86400)
        self._cache[key] = (value, time.time() + ttl)
```

# src/core/exceptions.py
class FMError(Exception):
    """基础异常类"""
    def __init__(self, message: str, error_code: str = "UNKNOWN_ERROR", details: Dict = None):
        self.message = message
        self.error_code = error_code
        self.details = details or {}
        super().__init__(self.message)

class AssetNotFoundError(FMError):
    """资产未找到异常"""
    pass

class DataFetchError(FMError):
    """数据获取异常"""
    pass

class ValidationError(FMError):
    """验证异常"""
    pass

# 统一的错误处理装饰器
def handle_errors(func):
    """统一错误处理装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        try:
            return func(*args, **kwargs)
        except FMError as e:
            # 记录业务异常
            logger.warning(f"Business error: {e.error_code} - {e.message}")
            raise
        except Exception as e:
            # 记录系统异常
            logger.error(f"System error in {func.__name__}: {str(e)}")
            raise FMError(
                message="系统内部错误",
                error_code="SYSTEM_ERROR",
                details={"function": func.__name__, "error": str(e)}
            )
    return wrapper
```

#### 问题: 缺乏监控指标
**建议方案**:
```python
# src/core/monitoring.py
import time
from typing import Dict, List
from dataclasses import dataclass
from datetime import datetime

@dataclass
class OperationMetric:
    """操作指标"""
    operation: str
    start_time: float
    end_time: float
    success: bool
    error: Optional[str] = None
    
    @property
    def duration(self) -> float:
        return self.end_time - self.start_time

class PerformanceMonitor:
    """性能监控器"""
    def __init__(self, max_metrics: int = 1000):
        self.metrics: List[OperationMetric] = []
        self.max_metrics = max_metrics
        
    def track_operation(self, operation: str):
        """跟踪操作装饰器"""
        def decorator(func):
            @wraps(func)
            def wrapper(*args, **kwargs):
                start_time = time.time()
                try:
                    result = func(*args, **kwargs)
                    self._add_metric(operation, start_time, time.time(), True)
                    return result
                except Exception as e:
                    self._add_metric(operation, start_time, time.time(), False, str(e))
                    raise
            return wrapper
        return decorator
    
    def _add_metric(self, operation: str, start_time: float, end_time: float, 
                   success: bool, error: Optional[str] = None):
        """添加指标"""
        metric = OperationMetric(operation, start_time, end_time, success, error)
        self.metrics.append(metric)
        
        # 保持指标数量在限制内
        if len(self.metrics) > self.max_metrics:
            self.metrics = self.metrics[-self.max_metrics:]
    
    def get_performance_report(self) -> Dict[str, Any]:
        """生成性能报告"""
        if not self.metrics:
            return {}
            
        successful = [m for m in self.metrics if m.success]
        failed = [m for m in self.metrics if not m.success]
        
        return {
            "total_operations": len(self.metrics),
            "success_rate": len(successful) / len(self.metrics) if self.metrics else 0,
            "average_duration": sum(m.duration for m in successful) / len(successful) if successful else 0,
            "operations_by_type": self._group_by_operation(),
            "recent_failures": [{"operation": m.operation, "error": m.error} 
                               for m in failed[-10:]]  # 最近10个失败
        }
    
    def _group_by_operation(self) -> Dict[str, Dict[str, Any]]:
        """按操作类型分组统计"""
        grouped = {}
        for metric in self.metrics:
            if metric.operation not in grouped:
                grouped[metric.operation] = {
                    "count": 0,
                    "success_count": 0,
                    "total_duration": 0,
                    "durations": []
                }
            
            group = grouped[metric.operation]
            group["count"] += 1
            if metric.success:
                group["success_count"] += 1
            group["total_duration"] += metric.duration
            group["durations"].append(metric.duration)
        
        # 计算统计量
        for operation, stats in grouped.items():
            if stats["durations"]:
                stats["avg_duration"] = stats["total_duration"] / len(stats["durations"])
                stats["p95_duration"] = sorted(stats["durations"])[int(len(stats["durations"]) * 0.95)]
            else:
                stats["avg_duration"] = 0
                stats["p95_duration"] = 0
            stats["success_rate"] = stats["success_count"] / stats["count"] if stats["count"] > 0 else 0
            del stats["durations"]  # 清理原始数据
        
        return grouped
```
